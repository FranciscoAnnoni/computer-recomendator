```typescript
import type { MetadataRoute } from "next";

// MetadataRoute.Sitemap — shape expected by app/sitemap.ts default export:
type Sitemap = Array<{
  url: string;                           // absolute URL
  lastModified?: Date | string;
  changeFrequency?: "always" | "hourly" | "daily" | "weekly" | "monthly" | "yearly" | "never";
  priority?: number;                     // 0.0 to 1.0
  alternates?: { languages?: Record<string, string> };
}>;

// MetadataRoute.Robots — shape expected by app/robots.ts default export:
type Robots = {
  rules: {
    userAgent: string | string[];
    allow?: string | string[];
    disallow?: string | string[];
    crawlDelay?: number;
  } | Array<{ userAgent: string | string[]; allow?: string | string[]; disallow?: string | string[] }>;
  sitemap?: string | string[];
  host?: string;
};
```
