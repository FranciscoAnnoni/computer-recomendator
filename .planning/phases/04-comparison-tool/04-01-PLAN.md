function useSheetSide(): "left" | "bottom" {
  const [side, setSide] = useState<"left" | "bottom">("bottom");
  useEffect(() => {
    const mq = window.matchMedia("(min-width: 768px)");
    setSide(mq.matches ? "left" : "bottom");
    const handler = (e: MediaQueryListEvent) => setSide(e.matches ? "left" : "bottom");
    mq.addEventListener("change", handler);
    return () => mq.removeEventListener("change", handler);
  }, []);
  return side;
}

    function removeSlot(index: number) {
      setSlots(prev => {
        const next = [...prev];
        next[index] = null;
        // If 3rd slot exists and becomes irrelevant, trim back to 2
        if (next.length === 3 && (next[0] === null || next[1] === null)) {
          return [next[0], next[1]];

        return next;
      });

    function handleSelectLaptop(laptop: Laptop) {

      setSlots(prev => {
        const next = [...prev];
        next[pickerSlotIndex] = laptop;

      });
      setPickerSlotIndex(null);
